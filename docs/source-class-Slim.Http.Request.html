 130     /**
 131      * Create new HTTP request with data extracted from the application
 132      * Environment object
 133      *
 134      * @param  Environment $environment The Slim application Environment
 135      *
 136      * @return static
 137      */
 138     public static function createFromEnvironment(Environment $environment)
 139     {
 140         $method = $environment['REQUEST_METHOD'];
 141         $uri = Uri::createFromEnvironment($environment);
 142         $headers = Headers::createFromEnvironment($environment);
 143         $cookies = Cookies::parseHeader($headers->get('Cookie', []));
 144         $serverParams = $environment->all();
 145         $body = new RequestBody();
 146         $uploadedFiles = UploadedFile::createFromEnvironment($environment);
 147 
 148         $request = new static($method, $uri, $headers, $cookies, $serverParams, $body, $uploadedFiles);
 149 
 150         if ($method === 'POST' &&
 151             in_array($request->getMediaType(), ['application/x-www-form-urlencoded', 'multipart/form-data'])
 152         ) {
 153             // parsed body must be $_POST
 154             $request = $request->withParsedBody($_POST);
 155         }
 156         return $request;
 157     }

 159     /**
 160      * Create new HTTP request.
 161      *
 162      * Adds a host header when none was provided and a host is defined in uri.
 163      *
 164      * @param string           $method        The request method
 165      * @param UriInterface     $uri           The request URI object
 166      * @param HeadersInterface $headers       The request headers collection
 167      * @param array            $cookies       The request cookies collection
 168      * @param array            $serverParams  The server environment variables
 169      * @param StreamInterface  $body          The request body object
 170      * @param array            $uploadedFiles The request uploadedFiles collection
 171      * @throws InvalidMethodException on invalid HTTP method
 172      */
 173     public function __construct(
 174         $method,
 175         UriInterface $uri,
 176         HeadersInterface $headers,
 177         array $cookies,
 178         array $serverParams,
 179         StreamInterface $body,
 180         array $uploadedFiles = []
 181     ) {
 182         try {
 183             $this->originalMethod = $this->filterMethod($method);
 184         } catch (InvalidMethodException $e) {
 185             $this->originalMethod = $method;
 186         }
 187 
 188         $this->uri = $uri;
 189         $this->headers = $headers;
 190         $this->cookies = $cookies;
 191         $this->serverParams = $serverParams;
 192         $this->attributes = new Collection();
 193         $this->body = $body;
 194         $this->uploadedFiles = $uploadedFiles;
 195 
 196         if (isset($serverParams['SERVER_PROTOCOL'])) {
 197             $this->protocolVersion = str_replace('HTTP/', '', $serverParams['SERVER_PROTOCOL']);
 198         }
 199 
 200         if (!$this->headers->has('Host') || $this->uri->getHost() !== '') {
 201             $this->headers->set('Host', $this->uri->getHost());
 202         }
 203 
 204         $this->registerMediaTypeParser('application/json', function ($input) {
 205             return json_decode($input, true);
 206         });
 207 
 208         $this->registerMediaTypeParser('application/xml', function ($input) {
 209             $backup = libxml_disable_entity_loader(true);
 210             $backup_errors = libxml_use_internal_errors(true);
 211             $result = simplexml_load_string($input);
 212             libxml_disable_entity_loader($backup);
 213             libxml_clear_errors();
 214             libxml_use_internal_errors($backup_errors);
 215             if ($result === false) {
 216                 return null;
 217             }
 218             return $result;
 219         });
 220 
 221         $this->registerMediaTypeParser('text/xml', function ($input) {
 222             $backup = libxml_disable_entity_loader(true);
 223             $backup_errors = libxml_use_internal_errors(true);
 224             $result = simplexml_load_string($input);
 225             libxml_disable_entity_loader($backup);
 226             libxml_clear_errors();
 227             libxml_use_internal_errors($backup_errors);
 228             if ($result === false) {
 229                 return null;
 230             }
 231             return $result;
 232         });
 233 
 234         $this->registerMediaTypeParser('application/x-www-form-urlencoded', function ($input) {
 235             parse_str($input, $data);
 236             return $data;
 237         });
 238 
 239         // if the request had an invalid method, we can throw it now
 240         if (isset($e) && $e instanceof InvalidMethodException) {
 241             throw $e;
 242         }
 243     }

 262     /**
 263      * Retrieves the HTTP method of the request.
 264      *
 265      * @return string Returns the request method.
 266      */
 267     public function getMethod()
 268     {
 269         if ($this->method === null) {
 270             $this->method = $this->originalMethod;
 271             $customMethod = $this->getHeaderLine('X-Http-Method-Override');
 272 
 273             if ($customMethod) {
 274                 $this->method = $this->filterMethod($customMethod);
 275             } elseif ($this->originalMethod === 'POST') {
 276                 $overrideMethod = $this->filterMethod($this->getParsedBodyParam('_METHOD'));
 277                 if ($overrideMethod !== null) {
 278                     $this->method = $overrideMethod;
 279                 }
 280 
 281                 if ($this->getBody()->eof()) {
 282                     $this->getBody()->rewind();
 283                 }
 284             }
 285         }
 286 
 287         return $this->method;
 288     }

 303      * Return an instance with the provided HTTP method.
 304      *
 305      * While HTTP method names are typically all uppercase characters, HTTP
 306      * method names are case-sensitive and thus implementations SHOULD NOT
 307      * modify the given string.
 308      *
 309      * This method MUST be implemented in such a way as to retain the
 310      * immutability of the message, and MUST return an instance that has the
 311      * changed request method.
 312      *
 313      * @param string $method Case-sensitive method.
 314      * @return static
 315      * @throws \InvalidArgumentException for invalid HTTP methods.
 316      */
 317     public function withMethod($method)
 318     {
 319         $method = $this->filterMethod($method);
 320         $clone = clone $this;
 321         $clone->originalMethod = $method;
 322         $clone->method = $method;
 323 
 324         return $clone;
 325     }

 327     /**
 328      * Validate the HTTP method
 329      *
 330      * @param  null|string $method
 331      * @return null|string
 332      * @throws \InvalidArgumentException on invalid HTTP method.
 333      */
 334     protected function filterMethod($method)
 335     {
 336         if ($method === null) {
 337             return $method;
 338         }
 339 
 340         if (!is_string($method)) {
 341             throw new InvalidArgumentException(sprintf(
 342                 'Unsupported HTTP method; must be a string, received %s',
 343                 (is_object($method) ? get_class($method) : gettype($method))
 344             ));
 345         }
 346 
 347         $method = strtoupper($method);
 348         if (!isset($this->validMethods[$method])) {
 349             throw new InvalidMethodException($this, $method);
 350         }
 351 
 352         return $method;
 353     }

 468     /**
 469      * Retrieves the message's request target.
 470      *
 471      * Retrieves the message's request-target either as it will appear (for
 472      * clients), as it appeared at request (for servers), or as it was
 473      * specified for the instance (see withRequestTarget()).
 474      *
 475      * In most cases, this will be the origin-form of the composed URI,
 476      * unless a value was provided to the concrete implementation (see
 477      * withRequestTarget() below).
 478      *
 479      * If no URI is available, and no request-target has been specifically
 480      * provided, this method MUST return the string "/".
 481      *
 482      * @return string
 483      */
 484     public function getRequestTarget()
 485     {
 486         if ($this->requestTarget) {
 487             return $this->requestTarget;
 488         }
 489 
 490         if ($this->uri === null) {
 491             return '/';
 492         }
 493 
 494         $basePath = $this->uri->getBasePath();
 495         $path = $this->uri->getPath();
 496         $path = $basePath . '/' . ltrim($path, '/');
 497 
 498         $query = $this->uri->getQuery();
 499         if ($query) {
 500             $path .= '?' . $query;
 501         }
 502         $this->requestTarget = $path;
 503 
 504         return $this->requestTarget;
 505     }

 508      * Return an instance with the specific request-target.

 510      * If the request needs a non-origin-form request-target  e.g., for
 511      * specifying an absolute-form, authority-form, or asterisk-form 
 512      * this method may be used to create an instance with the specified
 513      * request-target, verbatim.
 514      *

 516      * immutability of the message, and MUST return an instance that has the

 552     /**
 553      * Returns an instance with the provided URI.
 554      *
 555      * This method MUST update the Host header of the returned request by
 556      * default if the URI contains a host component. If the URI does not
 557      * contain a host component, any pre-existing Host header MUST be carried
 558      * over to the returned request.
 559      *
 560      * You can opt-in to preserving the original state of the Host header by
 561      * setting `$preserveHost` to `true`. When `$preserveHost` is set to
 562      * `true`, this method interacts with the Host header in the following ways:
 563      *
 564      * - If the the Host header is missing or empty, and the new URI contains
 565      *   a host component, this method MUST update the Host header in the returned
 566      *   request.
 567      * - If the Host header is missing or empty, and the new URI does not contain a
 568      *   host component, this method MUST NOT update the Host header in the returned
 569      *   request.
 570      * - If a Host header is present and non-empty, this method MUST NOT update
 571      *   the Host header in the returned request.
 572      *
 573      * This method MUST be implemented in such a way as to retain the
 574      * immutability of the message, and MUST return an instance that has the
 575      * new UriInterface instance.
 576      *
 577      * @link http://tools.ietf.org/html/rfc3986#section-4.3
 578      * @param UriInterface $uri New request URI to use.
 579      * @param bool $preserveHost Preserve the original state of the Host header.
 580      * @return static
 581      */
 582     public function withUri(UriInterface $uri, $preserveHost = false)
 583     {
 584         $clone = clone $this;
 585         $clone->uri = $uri;
 586 
 587         if (!$preserveHost) {
 588             if ($uri->getHost() !== '') {
 589                 $clone->headers->set('Host', $uri->getHost());
 590             }
 591         } else {
 592             if ($uri->getHost() !== '' && (!$this->hasHeader('Host') || $this->getHeaderLine('Host') === '')) {
 593                 $clone->headers->set('Host', $uri->getHost());
 594             }
 595         }
 596 
 597         return $clone;
 598     }

 641     {
 642         $contentType = $this->getContentType();
 643         $contentTypeParams = [];
 644         if ($contentType) {
 645             $contentTypeParts = preg_split('/\s*[;,]\s*/', $contentType);
 646             $contentTypePartsLength = count($contentTypeParts);
 647             for ($i = 1; $i < $contentTypePartsLength; $i++) {
 648                 $paramParts = explode('=', $contentTypeParts[$i]);
 649                 $contentTypeParams[strtolower($paramParts[0])] = $paramParts[1];
 650             }
 651         }
 652 
 653         return $contentTypeParams;
